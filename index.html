// Скрипт массового обновления уровней сложности async function massUpdateDifficulty() { console.log('🔄 Начинаю анализ и обновление тренажёров...'); // Динамически подгружаем нужные функции Firestore const { collection, getDocs, doc, updateDoc } = await import('https://www.gstatic.com/firebasejs/10.7.1/firebase-firestore.js'); const db = window.db; // Берем уже подключенную базу из вашего сайта // Ключевые слова для автоматического определения уровня (можете редактировать этот список) const keywords = { 'beginner': ['beginner', 'начальный', '1 класс', '2 класс', 'spotlight 2', 'spotlight 3', 'a1'], 'elementary': ['elementary', '3 класс', '4 класс', 'spotlight 4', 'spotlight 5', 'a2'], 'pre-intermediate': ['pre-intermediate', '5 класс', '6 класс', 'spotlight 6', 'spotlight 7', 'b1'], 'intermediate': ['intermediate', '7 класс', '8 класс', 'spotlight 8', 'spotlight 9', '1001 questions', 'b1+'], 'upper-intermediate': ['upper-intermediate', '9 класс', '10 класс', 'spotlight 10', 'oge', 'огэ', 'b2'], 'advanced': ['advanced', '11 класс', 'егэ', 'ege', 'c1', 'c2'] }; const snap = await getDocs(collection(db, 'trainers')); let updatedCount = 0; for (const d of snap.docs) { const data = d.data(); // Объединяем название и описание для поиска ключевых слов const text = (data.title + ' ' + (data.description || '')).toLowerCase(); let newDiff = data.difficulty || 'intermediate'; // Уровень по умолчанию, если не найдено // Ищем совпадения for (const [level, words] of Object.entries(keywords)) { if (words.some(w => text.includes(w))) { newDiff = level; break; } } // Обновляем только если уровень изменился или его не было if (data.difficulty !== newDiff) { await updateDoc(doc(db, 'trainers', d.id), { difficulty: newDiff }); console.log(`✅ Обновлено: "${data.title}" → ${newDiff}`); updatedCount++; } } console.log(`🎉 Готово! Успешно обновлено ${updatedCount} тренажёров. Перезагрузите страницу, чтобы увидеть изменения.`); } // Запуск функции massUpdateDifficulty();
